  3. {{T .Lang "Name: Importinvoices"}}

    "importinvoices": {
      "command": "C:\\path\\to\\importinvoices\\importinvoices.cmd",
      "args": ["mcp", "--auth-token", "YOUR_TOKEN_HERE"]
    }
  }
